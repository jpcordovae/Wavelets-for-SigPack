77  parser(const std::string& fname)
78  {
79  std::ifstream fh;
80  std::string line;
81  size_t mark = 0;
82 
83  // Clear parameter map
84  par_map.clear();
85 
86  // Open file
87  fh.open(fname.c_str());
88  if (!fh)
89  {
90  wrn_handler("Could not find " + fname);
91  }
92  else
93  {
94  // Parse
95  while (std::getline(fh, line))
96  {
97  std::string keyS="";
98  std::string dataS="";
99 
100  // Skip empty lines
101  if (line.empty())
102  continue;
103 
104  // Skip lines with only whitespace
105  if(line.find_first_not_of("\t ")==std::string::npos)
106  continue;
107 
108  // Remove comment
109  mark = line.find("%");
110  if(mark!=std::string::npos)
111  line.erase(mark,line.length());
112 
113  // Do we have a '='
114  mark = line.find("=");
115  if(mark!=std::string::npos)
116  {
117  // Find key
118  keyS = line.substr(line.find_first_not_of("\t "),mark-line.find_first_not_of("\t "));
119  keyS = keyS.substr(0,keyS.find_last_not_of("\t ")+1);
120 
121  // Find data
122  dataS = line.substr(mark+1,line.length());
123  dataS = dataS.substr(0,dataS.find_last_not_of("\t ")+1);
124 
125  // Do we have a string
126  mark = dataS.find("\"");
127  if(mark!=std::string::npos)
128  {
129  dataS = dataS.substr(mark+1,dataS.length());
130  dataS = dataS.substr(0,dataS.find_last_of("\""));
131  }
132  // Do we have a vector/matrix
133  mark = dataS.find("[");
134  if(mark!=std::string::npos)
135  {
136  dataS = dataS.substr(mark+1,dataS.length());
137  dataS = dataS.substr(0,dataS.find_last_of("]"));
138  }
139 
140  // Insert to map
141  par_map.insert(std::pair<std::string, std::string>(keyS, dataS));
142  }
143  }
144 
145  // Close file
146  fh.close();
147  }
148  }
