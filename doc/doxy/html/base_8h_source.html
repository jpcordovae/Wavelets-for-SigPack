103  arma_inline arma::vec unwrap( const arma::vec& x )
104  {
105  arma::vec P;
106  double pacc = 0, pdiff = 0;
107  const double thr=PI*170/180;
108  P.copy_size(x);
109  P(0)=x(0);

111  {
112  pdiff = x(r)-x(r-1);

115  P(r)=pacc+x(r);
