76  arma_inline void lti2discr(const arma::mat& F,const arma::mat& W, const arma::mat& Qc, const double dT, arma::mat& A,arma::mat& Q)
77  {
78  arma::uword M = F.n_rows;
79 
80  // Solve A
81  A = arma::expmat(F*dT);
82 
83  // Solve Q by using matrix fraction decomposition
84  arma::mat AB = arma::zeros(2*M,M);
85  arma::mat CD = arma::zeros(2*M,2*M);
86  arma::mat EF = arma::zeros(2*M,M);
87  EF.submat(M,0, 2*M-1,M-1) = arma::eye(M,M);
88  CD.submat(0,0, M-1,M-1) = F;
89  CD.submat(M,M,2*M-1,2*M-1) = -F.t();
90  CD.submat(0,M,M-1,2*M-1) = W*Qc*W.t();
91 
92  AB = arma::expmat(CD*dT)*EF;
93 
94  Q = AB.rows(0,M-1)*arma::inv(AB.rows(M,2*M-1));
95  }

582  {

584  arma::mat y(Ny,1);
585  S.set_size(Ny,Ny);
586  C.set_size(N,Ny);
587
