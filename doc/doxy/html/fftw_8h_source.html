226  void fft2( arma::mat& x, arma::cx_mat& Pxx)
227  {
228  arma::cx_mat Ptmp(R / 2 + 1, C,arma::fill::ones);
229  double* in = x.memptr();
230  fftw_complex* out = reinterpret_cast<fftw_complex*>(Ptmp.memptr());
231  if (pl_fft == NULL)
232  {
233  pl_fft = fftw_plan_dft_r2c_2d( C,R, in, out, alg); // Column to row-major order trick: switch C and R
234  if (pl_fft == NULL)
235  {
236  err_handler("Unable to create real data FFTW plan");
237  }
238  }
239 
240  fftw_execute_dft_r2c(pl_fft, in, out);
241 
242  // Reshape Pxx - upper half
243  std::complex<double>* ptr= reinterpret_cast<std::complex<double>*>(out);
244  const unsigned int Roff = R / 2 + 1;
245  for (unsigned int r = 0; r < Roff; r++)
246  {
247  for (unsigned int c = 0; c < C; c++)
248  {
249  Pxx(r, c) = ptr[r + c*Roff];
250  }
251  }
252  // Reshape Pxx - conj symmetry
253  for (unsigned int r = Roff; r < R; r++)
254  {
255  Pxx(r, 0) = conj(ptr[R - r]);
256  for (unsigned int c = 1; c < C; c++)
257  {
258  Pxx(r, c) = conj(ptr[R - r + (C - c)*Roff]);
259  }
260  }
261  }
